<script type="module">

import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.0/firebase-app.js";

import {

  getFirestore,

  collection,

  getDocs

} from "https://www.gstatic.com/firebasejs/10.7.0/firebase-firestore.js";


const firebaseConfig = {

  apiKey: "YOUR_KEY",

  authDomain: "YOUR_DOMAIN",

  projectId: "YOUR_ID"

};


const app = initializeApp(firebaseConfig);

const db = getFirestore(app);


let cart = [];


// 讀商品

window.loadProducts = async function(){

  let snap = await getDocs(collection(db,"products"));


  let html = "";

  snap.forEach(doc=>{

    let p = doc.data();


    html += `

      <div class="card">

        <b>${p.name}</b><br>

        $${p.price} / 庫存:${p.stock}

        <button onclick="add('${p.name}',${p.price})">加入</button>

      </div>

    `;

  });


  products.innerHTML = html;

}


// 加入購物車

window.add = function(name,price){

  cart.push({name,price});

  render();

}


// 結帳（呼叫後端）

window.checkout = async function(){


  let res = await fetch("/checkout",{

    method:"POST",

    headers:{ "Content-Type":"application/json" },

    body: JSON.stringify({ cart })

  });


  let data = await res.json();


  alert("發票號：" + data.invoiceNo);


  cart = [];

  render();

  loadProducts();

}


function render(){

  cartDiv.innerHTML = cart.map(c=>`${c.name} $${c.price}`).join("<br>");


  total.innerText = cart.reduce((a,b)=>a+b.price,0);

}


window.loadProducts();


</script>

<button class="green" onclick="checkout()">結帳</button>

</div>
